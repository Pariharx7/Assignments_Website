Write a JavaScript program demonstrating the use of various mathematical functions.


<!DOCTYPE html>
<html lang="en">
<head>
<title>Question 25</title>
</head>
<body>
<script>

function myFunction() {

// abs()
document.write("The absolute value of -5 is: " + Math.abs(-5) +"<br>");

// ceil()
document.write("The smallest integer greater than or equal to 3.14 is: " + Math.ceil(3.14) +"<br>");

// floor()
document.write("The largest integer less than or equal to 3.14 is: " + Math.floor(3.14) +"<br>");

// max()
document.write("The largest of 5 and 10 is: " + Math.max(5, 10) +"<br>");

// min()
document.write("The smallest of 5 and 10 is: " + Math.min(5,10) +"<br>");

//pow()
document.write("The 5 raised to power 2 is: " + Math.pow(5,2) +"<br>");

//sqrt()
document.write("The square root of 16 is: " + Math.sqrt(16) +"<br>");

}


myFunction();

</script>
